fs-input-validation

Manual Validation

This is valid

This is invalid

Code

    
      <fs-input-validation id='v1'>
        <input type="text" />
        <p valid-text>This is valid</p>
        <p invalid-text>This is invalid</p>
      </fs-input-validation>
      <script>
        var input = document.querySelector('#i1');
        var validation = document.querySelector('#v1');

        input.addEventListener('input', function() {
          validation.status = /^[a-zA-Z]+$/.test(input.value) ? 'valid' : 'invalid';
        });
      </script>

Code

    
      <fs-input-validation id='v2'>
        <input type="text" id='i2' required pattern='^[0-9]+$'/>
        <p valid-text>This is valid</p>
        <p invalid-text>This is invalid</p>
      </fs-input-validation>
      <script>
        var input2 = document.querySelector('#i2');
        var validation2 = document.querySelector('#v2');

        input2.addEventListener('input', function() {
          validation2.status = input2.checkValidity() ? 'valid' : 'invalid';
        });
      </script>
